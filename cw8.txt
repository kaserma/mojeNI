using System;

class Program
{
    // Funkcja do obliczania NWD dwóch liczb
    static int NWD(int a, int b)
    {
        while (b != 0)
        {
            int temp = b;
            b = a % b;
            a = temp;
        }
        return a;
    }

    static void Main()
    {
        Console.Write("Podaj dolny zakres: ");
        int lowerLimit = int.Parse(Console.ReadLine());

        Console.Write("Podaj górny zakres: ");
        int upperLimit = int.Parse(Console.ReadLine());

        Random rand = new Random();

        Console.WriteLine("Pary liczb względnie pierwszych z podanego zakresu:");

        for (int i = lowerLimit; i <= upperLimit; i++)
        {
            for (int j = i + 1; j <= upperLimit; j++)
            {
                if (NWD(i, j) == 1)
                {
                    Console.WriteLine($"Znaleziona para liczb względnie pierwszych: {i} i {j}");
                }
            }
        }
    }
}
